def initialize
  super
  @global_time = BBSTime.new
  @items = {}
  @variables = []
  @constraints = {}
  # leave like this for infinite environment
  # infinite environment may cause slow calculations
  @constraints[:min] = CVector.new(0, 0, 0, 1)
  @constraints[:max] = CVector.new(0, 0, 0, 1)
  @name = ''
end

  f.close
end
