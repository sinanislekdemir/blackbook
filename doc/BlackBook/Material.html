def start_render
  GL.Enable(GL::COLOR_MATERIAL)
  GL.Enable(GL::BLEND)
  GL.BlendFunc(GL::SRC_ALPHA, GL::ONE_MINUS_SRC_ALPHA)
  GL.Color4f(@color.x, @color.y, @color.z, @color.w)
end
